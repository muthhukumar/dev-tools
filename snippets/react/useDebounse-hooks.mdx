---
title: useDebounce custom hook
description: Debounce the given value with the given delay
---

```ts
import { useRef, useState, useEffect } from 'react'

export function useDebounce<ValueType>(value: ValueType, delay: number): ValueType | null {
  const [debouncedValue, setDebouncedValue] = useState<null | ValueType>(null)

  const previousValueRef = useRef<ValueType | undefined>()

  useEffect(() => {
    if (value !== previousValueRef.current) {
      setDebouncedValue(null)
    }
  }, [value])

  useEffect(() => {
    const handler = setTimeout(() => {
      setDebouncedValue(value)
      previousValueRef.current = value
    }, delay)

    return () => {
      clearTimeout(handler)
      previousValueRef.current = undefined
    }
  }, [value, delay])

  return debouncedValue
}
```

This hook accept value and a delay as the parameter. When the value parameter is changed then this
hook will return the debounce value as null and after the given delay, it will return the given value.
